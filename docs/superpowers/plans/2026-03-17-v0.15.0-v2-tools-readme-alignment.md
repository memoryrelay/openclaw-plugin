```typescript
  // --------------------------------------------------------------------------
  // V2 Async API Methods (v0.15.0)
  // --------------------------------------------------------------------------

  async storeAsync(
    content: string,
    metadata?: Record<string, string>,
    project?: string,
    importance?: number,
    tier?: string,
  ): Promise<{ id: string; status: string; job_id: string; estimated_completion_seconds: number }> {
    if (!content || content.length === 0 || content.length > 50000) {
      throw new Error("Content must be between 1 and 50,000 characters");
    }
    const body: Record<string, unknown> = {
      content,
      agent_id: this.agentId,
    };
    if (metadata) body.metadata = metadata;
    if (project) body.project = project;
    if (importance != null) body.importance = importance;
    if (tier) body.tier = tier;
    return this.request("POST", "/v2/memories", body);
  }

  async getMemoryStatus(memoryId: string): Promise<{
    id: string;
    status: "pending" | "processing" | "ready" | "failed";
    created_at: string;
    updated_at: string;
    error?: string;
  }> {
    return this.request("GET", `/v2/memories/${memoryId}/status`);
  }

  async buildContextV2(
    query: string,
    options?: {
      maxMemories?: number;
      maxTokens?: number;
      aiEnhanced?: boolean;
      searchMode?: "semantic" | "hybrid" | "keyword";
      excludeMemoryIds?: string[];
    },
  ): Promise<any> {
    const body: Record<string, unknown> = {
      query,
      agent_id: this.agentId,
    };
    if (options?.maxMemories != null) body.max_memories = options.maxMemories;
    if (options?.maxTokens != null) body.max_tokens = options.maxTokens;
    if (options?.aiEnhanced != null) body.ai_enhanced = options.aiEnhanced;
    if (options?.searchMode) body.search_mode = options.searchMode;
    if (options?.excludeMemoryIds) body.exclude_memory_ids = options.excludeMemoryIds;
    return this.request("POST", "/v2/context", body);
  }
```

```typescript
  // /memory-context — Build context bundle (v0.15.0)
  api.registerCommand?.({
    name: "memory-context",
    description: "Build a ranked context bundle from memories (usage: /memory-context <query> [--max-memories N] [--max-tokens N] [--ai-enhanced] [--search-mode semantic|hybrid|keyword])",
    acceptsArgs: true,
    requireAuth: true,
    handler: async (ctx) => {
      try {
        const { positional, flags } = parseCommandArgs(ctx.args);
        const query = positional.join(" ");
        if (!query) {
          return { text: "Usage: /memory-context <query> [--max-memories N] [--max-tokens N] [--ai-enhanced] [--search-mode semantic|hybrid|keyword]" };
        }

        const maxMemories = flags["max-memories"] ? parseInt(String(flags["max-memories"]), 10) : undefined;
        const maxTokens = flags["max-tokens"] ? parseInt(String(flags["max-tokens"]), 10) : undefined;
        const aiEnhanced = flags["ai-enhanced"] === true;
        const searchMode = flags["search-mode"] ? String(flags["search-mode"]) as "semantic" | "hybrid" | "keyword" : undefined;

        const result = await client.buildContextV2(query, {
          maxMemories,
          maxTokens,
          aiEnhanced,
          searchMode,
        });

        const memories = result?.context || [];
        if (!Array.isArray(memories) || memories.length === 0) {
          return { text: `No relevant memories found for "${query}".` };
        }

        const lines: string[] = [
          `Context Bundle for "${query}"`,
          "━".repeat(60),
        ];

        if (result.total_tokens) {
          lines.push(`Tokens: ${result.total_tokens} | Memories: ${memories.length}`);
          lines.push("");
        }

        for (const item of memories) {
          const m = item.memory || item;
          const content = String(m.content || "").slice(0, 120);
          const score = item.score != null ? ` (${(item.score * 100).toFixed(1)}%)` : "";
          const category = m.metadata?.category ? ` [${m.metadata.category}]` : "";
          lines.push(`${content}${content.length >= 120 ? "..." : ""}${score}${category}`);
          lines.push("");
        }

        if (result.summary) {
          lines.push("─".repeat(60));
          lines.push("AI Summary:");
          lines.push(result.summary);
        }

        lines.push("─".repeat(60));
        lines.push(`${memories.length} memory/memories included`);
        return { text: lines.join("\n") };
      } catch (err) {
        return { text: `Error: ${String(err)}`, isError: true };
      }
    },
  });
```
