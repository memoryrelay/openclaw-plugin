// Since parseCommandArgs is not exported, we test it indirectly through command
// handlers. But we can replicate the function here for unit testing.

function parseCommandArgs(input: string | undefined): { positional: string[]; flags: Record<string, string | boolean> } {
  const positional: string[] = [];
  const flags: Record<string, string | boolean> = {};

  if (!input || input.trim() === "") {
    return { positional, flags };
  }

  const tokens: string[] = [];
  let current = "";
  let inQuote: string | null = null;

  for (const ch of input) {
    if (inQuote) {
      if (ch === inQuote) {
        inQuote = null;
      } else {
        current += ch;
      }
    } else if (ch === '"' || ch === "'") {
      inQuote = ch;
    } else if (ch === " " || ch === "\t") {
      if (current) {
        tokens.push(current);
        current = "";
      }
    } else {
      current += ch;
    }
  }
  if (current) tokens.push(current);

  let i = 0;
  while (i < tokens.length) {
    const token = tokens[i];
    if (token.startsWith("--")) {
      const key = token.slice(2);
      const next = tokens[i + 1];
      if (next && !next.startsWith("--")) {
        flags[key] = next;
        i += 2;
      } else {
        flags[key] = true;
        i += 1;
      }
    } else {
      positional.push(token);
      i += 1;
    }
  }

  return { positional, flags };
}

describe("parseCommandArgs", () => {
  test("should return empty for undefined input", () => {
    const result = parseCommandArgs(undefined);
    expect(result).toEqual({ positional: [], flags: {} });
  });

  test("should return empty for empty string", () => {
    const result = parseCommandArgs("");
    expect(result).toEqual({ positional: [], flags: {} });
  });

  test("should parse positional arguments", () => {
    const result = parseCommandArgs("hello world");
    expect(result.positional).toEqual(["hello", "world"]);
    expect(result.flags).toEqual({});
  });

  test("should parse flags with values", () => {
    const result = parseCommandArgs("--limit 10 --project my-api");
    expect(result.positional).toEqual([]);
    expect(result.flags).toEqual({ limit: "10", project: "my-api" });
  });

  test("should parse boolean flags", () => {
    const result = parseCommandArgs("--active --verbose");
    expect(result.flags).toEqual({ active: true, verbose: true });
  });

  test("should parse mixed positional and flags", () => {
    const result = parseCommandArgs("authentication --limit 5 --project my-api");
    expect(result.positional).toEqual(["authentication"]);
    expect(result.flags).toEqual({ limit: "5", project: "my-api" });
  });

  test("should handle quoted strings", () => {
    const result = parseCommandArgs('"hello world" --limit 10');
    expect(result.positional).toEqual(["hello world"]);
    expect(result.flags).toEqual({ limit: "10" });
  });

  test("should handle single-quoted strings", () => {
    const result = parseCommandArgs("'deploy to prod' --project api");
    expect(result.positional).toEqual(["deploy to prod"]);
    expect(result.flags).toEqual({ project: "api" });
  });

  test("should handle flag followed by another flag", () => {
    const result = parseCommandArgs("--active --limit 5");
    expect(result.flags).toEqual({ active: true, limit: "5" });
  });

  test("should handle flag at end of input as boolean", () => {
    const result = parseCommandArgs("--limit 10 --active");
    expect(result.flags).toEqual({ limit: "10", active: true });
  });
});

  // /memory-search — Semantic memory search (v0.14.0)
  api.registerCommand?.({
    name: "memory-search",
    description: "Search memories semantically with filters (usage: /memory-search <query> [--limit N] [--project slug] [--threshold N])",
    acceptsArgs: true,
    requireAuth: true,
    handler: async (ctx) => {
      try {
        const { positional, flags } = parseCommandArgs(ctx.args);
        const query = positional.join(" ");
        if (!query) {
          return { text: "Usage: /memory-search <query> [--limit N] [--project slug] [--threshold N]" };
        }

        const limit = flags.limit ? parseInt(String(flags.limit), 10) : 10;
        const threshold = flags.threshold ? parseFloat(String(flags.threshold)) : 0.3;
        const project = flags.project ? String(flags.project) : undefined;

        const results = await client.search(query, limit, threshold, { project });

        if (!results || results.length === 0) {
          return { text: `No memories found matching "${query}".` };
        }

        const lines: string[] = [
          `Search Results for "${query}"`,
          "━".repeat(60),
        ];

        for (const result of results) {
          const m = result.memory || result;
          const content = String(m.content || "").slice(0, 120);
          const score = result.score != null ? ` (${(result.score * 100).toFixed(1)}%)` : "";
          const category = m.metadata?.category ? ` [${m.metadata.category}]` : "";
          const date = m.created_at
            ? new Date(typeof m.created_at === "number" ? m.created_at : Date.parse(m.created_at)).toLocaleDateString()
            : "";

          lines.push(`${content}${content.length >= 120 ? "..." : ""}`);
          lines.push(`  ID: ${m.id}${score}${category} ${date}`);
          lines.push("");
        }

        lines.push(`${results.length} result(s) found`);
        return { text: lines.join("\n") };
      } catch (err) {
        return { text: `Error: ${String(err)}`, isError: true };
      }
    },
  });

  api.registerCommand?.({
    name: "memory-validate",
    description: "Run production readiness checks and show a health grade",
    requireAuth: true,
    handler: async (_ctx) => {
      const checks: Array<{ name: string; status: "PASS" | "FAIL" | "WARN"; detail: string }> = [];

      // 1. API connectivity
      try {
        await client.health();
        checks.push({ name: "API Connectivity", status: "PASS", detail: `Endpoint: ${apiUrl}` });
      } catch (err) {
        checks.push({ name: "API Connectivity", status: "FAIL", detail: String(err) });
      }

      // 2. API health
      try {
        const health = await client.health();
        const healthStatus = String(health.status).toLowerCase();
        const isHealthy = VALID_HEALTH_STATUSES.includes(healthStatus);
        checks.push({
          name: "API Health",
          status: isHealthy ? "PASS" : "WARN",
          detail: `Status: ${health.status}`,
        });
      } catch (err) {
        checks.push({ name: "API Health", status: "FAIL", detail: String(err) });
      }

      // 3. Core tools registered (TOOL_GROUPS is Record<string, string[]>)
      const coreTools = ["memory_store", "memory_recall", "memory_list"];
      const registeredTools = Object.values(TOOL_GROUPS).flat();
      const missingTools = coreTools.filter((t) => !registeredTools.includes(t));
      checks.push({
        name: "Core Tools",
        status: missingTools.length === 0 ? "PASS" : "FAIL",
        detail: missingTools.length === 0 ? `${coreTools.length} core tools registered` : `Missing: ${missingTools.join(", ")}`,
      });

      // 4. Auto-recall
      const autoRecallEnabled = cfg?.autoRecall ?? true;
      checks.push({
        name: "Auto-Recall",
        status: autoRecallEnabled ? "PASS" : "WARN",
        detail: autoRecallEnabled ? "Enabled" : "Disabled (enable for zero-friction experience)",
      });

      // 5. Auto-capture
      checks.push({
        name: "Auto-Capture",
        status: autoCaptureConfig.enabled ? "PASS" : "WARN",
        detail: autoCaptureConfig.enabled ? `Enabled (tier: ${autoCaptureConfig.tier})` : "Disabled",
      });

      // 6. Memory count
      try {
        await client.list(1);
        checks.push({
          name: "Memory Storage",
          status: "PASS",
          detail: "Can read memories",
        });
      } catch (err) {
        checks.push({ name: "Memory Storage", status: "FAIL", detail: String(err) });
      }

      // 7. Agent ID configured
      checks.push({
        name: "Agent ID",
        status: agentId && agentId !== "default" ? "PASS" : "WARN",
        detail: agentId ? `Agent: ${agentId}` : "Not configured (using default)",
      });

      // Format output
      const lines: string[] = ["MemoryRelay Production Readiness", "━".repeat(50)];
      let passed = 0;
      let failed = 0;
      let warned = 0;

      for (const check of checks) {
        const symbol = check.status === "PASS" ? "[PASS]" : check.status === "FAIL" ? "[FAIL]" : "[WARN]";
        lines.push(`${symbol} ${check.name}: ${check.detail}`);
        if (check.status === "PASS") passed++;
        else if (check.status === "FAIL") failed++;
        else warned++;
      }

      lines.push("─".repeat(50));

      const total = checks.length;
      let grade: string;
      if (passed === total) grade = "A+";
      else if (passed >= total - 1) grade = "A";
      else if (passed >= total - 2) grade = "B+";
      else if (passed >= total - 3) grade = "B";
      else grade = "F";

      const productionReady = failed === 0;

      lines.push(`Summary: ${passed}/${total} passed, ${failed} failed, ${warned} warnings`);
      lines.push(`Grade: ${grade}`);
      lines.push(`Production Ready: ${productionReady ? "Yes" : "No"}`);

      return { text: lines.join("\n") };
    },
  });

  // /memory-sessions — List sessions (v0.14.0)
  api.registerCommand?.({
    name: "memory-sessions",
    description: "List active and recent sessions (usage: /memory-sessions [--limit N] [--status active|ended] [--active] [--project slug])",
    acceptsArgs: true,
    requireAuth: true,
    handler: async (ctx) => {
      try {
        const { flags } = parseCommandArgs(ctx.args);
        const limit = flags.limit ? parseInt(String(flags.limit), 10) : 10;
        const project = flags.project ? String(flags.project) : undefined;
        let status = flags.status ? String(flags.status) : undefined;
        if (flags.active === true) status = "active";

        const result = await client.listSessions(limit, project, status);
        const sessions = result?.data || result || [];

        if (!Array.isArray(sessions) || sessions.length === 0) {
          return { text: "No sessions found." };
        }

        const lines: string[] = [
          "MemoryRelay Sessions",
          "━".repeat(70),
        ];

        for (const s of sessions) {
          const id = s.id || s.session_id || "unknown";
          const sessionStatus = s.status || "unknown";
          const startedAt = s.started_at || s.created_at;
          const startDate = startedAt
            ? new Date(typeof startedAt === "number" ? startedAt : Date.parse(startedAt)).toLocaleString()
            : "unknown";
          const summary = s.summary ? String(s.summary).slice(0, 80) : "";

          let duration = "ongoing";
          if (s.ended_at && startedAt) {
            const startMs = typeof startedAt === "number" ? startedAt : Date.parse(startedAt);
            const endMs = typeof s.ended_at === "number" ? s.ended_at : Date.parse(s.ended_at);
            const durationMin = Math.round((endMs - startMs) / 60000);
            duration = `${durationMin}m`;
          }

          lines.push(`[${sessionStatus.toUpperCase()}] ${id}`);
          lines.push(`  Started: ${startDate} | Duration: ${duration}`);
          if (summary) lines.push(`  Summary: ${summary}${summary.length >= 80 ? "..." : ""}`);
          lines.push("");
        }

        lines.push(`${sessions.length} session(s)`);
        return { text: lines.join("\n") };
      } catch (err) {
        return { text: `Error: ${String(err)}`, isError: true };
      }
    },
  });

  api.registerCommand?.({
    name: "memory-forget",
    description: "Delete a memory by ID (usage: /memory-forget <memory-id>)",
    acceptsArgs: true,
    requireAuth: true,
    handler: async (ctx) => {
      const { positional } = parseCommandArgs(ctx.args);
      const memoryId = positional[0];
      if (!memoryId) {
        return { text: "Usage: /memory-forget <memory-id>\n\nUse /memory-search to find memory IDs." };
      }

      try {
        // Fetch memory first to show what was deleted
        let contentPreview = "";
        try {
          const memory = await client.get(memoryId);
          contentPreview = String(memory.content || "").slice(0, 100);
        } catch (_) {
          // Memory may not be fetchable, proceed with delete anyway
        }

        await client.delete(memoryId);

        if (contentPreview) {
          return { text: `Deleted memory ${memoryId}: "${contentPreview}${contentPreview.length >= 100 ? "..." : ""}"` };
        }
        return { text: `Deleted memory ${memoryId}.` };
      } catch (err) {
        const errStr = String(err);
        if (errStr.includes("not found") || errStr.includes("404")) {
          return { text: `Memory ${memoryId} not found.` };
        }
        return { text: `Error: ${errStr}`, isError: true };
      }
    },
  });
